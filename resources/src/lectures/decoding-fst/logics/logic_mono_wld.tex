
\newcommand{\wldadjcond}{
\begin{array}{l }
i = l\\
\end{array}
}

\newcommand{\wldnonadjcond}{
\begin{array}{l }
l < i \leq I \\
\delta(i, l) \leq d \\
c_{i-l} = \bzero \\
\end{array}
}
