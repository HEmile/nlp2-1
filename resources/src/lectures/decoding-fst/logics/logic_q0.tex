
\newcommand{\wldadjside}{
\begin{array}{l }
i = l \\
\bigoplus_{k=i-l}^{i'-l} c_k = \bzero \\
\end{array}
}

\newcommand{\wldadjcons}{
\begin{array}{l l }
%\\
\text{where} & l' = l + \alpha_1(C) + 1 \\
& C' \ll \alpha_1(C) + 1
\end{array}
}
\newcommand{\wldnadjside}{
\begin{array}{l }
i > l \\
\bigoplus_{k=i-l}^{i'-l} c_k = \bzero \\
|r-i+1| \leq d \\
|i'-l+1| \leq d\\
\end{array}
}
\newcommand{\wldnadjcons}{
\begin{array}{l l }
%\\
\text{where} & c_k' = c_k \text{ if } k < i-l \text{ or } k > i'-l \text{ else } \bone 
\end{array}
}
